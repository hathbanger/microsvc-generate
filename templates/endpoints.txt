// MakeBarEndpoint - returns endpoint to bar function
func MakeBarEndpoint(
	s Service,
	l log.Logger,
	mw []endpoint.Middleware,
) endpoint.Endpoint {

	var e endpoint.Endpoint
	e = makeBarEndpoint(s)
	// add middlewares
	for _, m := range mw {
		e = m(e)
	}
	e = LoggingMiddleware(log.With(l, "method", "Bar"))(e)
	return e
}

func makeBarEndpoint(s Service) endpoint.Endpoint {
	return func(ctx context.Context, request interface{}) (interface{}, error) {
		req, ok := request.(models.BarRequest)
		if !ok {
			return nil, ErrBadRequest
		}
		return s.Bar(ctx, req)
	}
}
